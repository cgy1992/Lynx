uniform sampler2D normalMap;
uniform sampler2D tex;
uniform sampler2DShadow ShadowMap;
varying vec4 ShadowCoord;

varying vec3 lightDir;

void main()
{
    vec3 n = normalize(texture2D(normalMap, gl_TexCoord[0].st).xyz * 2.0 - 1.0);
    vec3 l = normalize(lightDir);
    float nDotL = max(0.0, dot(n, l));

    float shadow = shadow2DProj(ShadowMap, ShadowCoord).r;

    // gl_FragColor = color * texture2D(tex, gl_TexCoord[0].st);
    gl_FragColor = (0.6 + nDotL*0.4) * texture2D(tex, gl_TexCoord[0].st);
}
